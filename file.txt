apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: datamodels-staging-ingress
  namespace: datamodels-staging
  annotations:
    kubernetes.io/ingress.class: "nginx"
spec:
  tls:
  - hosts:
    - datamodels-staging.imsglobal.org  # Replace with your actual domain
    secretName: datamodels-staging-tls  # The secret created by Cert-Manager
  rules:
  - host: datamodels-staging.imsglobal.org
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: datamodels-staging  # Your service name
            port:
              number: 8080  # Make sure this matches your service's target port
