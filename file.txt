# Set the base directory from the environment variable
$runDir = $env:RUN_DIR
if (-not $runDir) {
    Write-Output "Environment variable RUN_DIR is not set."
    return
}

# Define the source and destination paths
$batchScripts = Join-Path -Path $runDir -ChildPath "batch-scripts"
$batchScriptsBackup = Join-Path -Path $runDir -ChildPath "batch-scripts-backup"

# Check if the source directory exists
if (-not (Test-Path -Path $batchScripts)) {
    Write-Output "Source directory '$batchScripts' does not exist."
    return
}

# Ensure the backup directory exists, create it if it does not
if (-not (Test-Path -Path $batchScriptsBackup)) {
    New-Item -ItemType Directory -Path $batchScriptsBackup | Out-Null
}

try {
    # Copy contents from BATCH-SCRIPTS to BATCH-SCRIPTS-BACKUP with overwrite
    Copy-Item -Path $batchScripts\* -Destination $batchScriptsBackup -Recurse -Force
    Write-Output "Contents copied successfully from '$batchScripts' to '$batchScriptsBackup'."
}
catch {
    Write-Output "An error occurred during copying: $_"
}
